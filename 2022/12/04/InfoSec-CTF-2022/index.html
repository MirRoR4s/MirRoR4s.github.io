from Crypto.Util.number import getPrime, GCD, bytes_to_long


class RSA:

    def __init__(self, ll: int) -> None:
        self.e = 65537
        self.p, self.q = getPrime(ll // 2), getPrime(ll // 2)
        while GCD(self.e, self.p - 1) != 1 or GCD(self.e, self.q - 1) != 1 or self.p == self.q:
            self.p, self.q = getPrime(ll // 2), getPrime(ll // 2)
        self.d = pow(self.e, -1, (self.p - 1) * (self.q - 1))
        self.n = self.p * self.q

    def enc(self, x: int) -> int:
        return pow(x, self.e, self.n)

    def dec(self, y: int) -> int:
        return pow(y, self.d, self.n)

已知flag的前五位是 flag{，可以利用已知明文攻击获得密钥的前五个字节。然后我们爆破密钥的第六个字节（共有256种可能），这样就得到了密钥的前六个字节。然后我们根据明文的第六个到第十一个字节是和密钥的前六个字节异或的特点可以恢复明文的六到十一个字节。恢复这部分明文之后同理就可以拿到密钥的六到十一字节。

QR code

题目描述

Crypto

Alice decided to send Bob a QR code. To do this, she generated a key using a physical random number generator, the length of which did not exceed 50,000 bytes. Next, she performed a byte-by-byte xor operation of QR code with a repeating key. Can you recover the QR code?

解题步骤

  1. 注意到加密的QR code 文件的格式是 .bmp，如果你以这种格式保存寻常的黑白 QR code并在一个十六进制编辑器种解释其内容，你就能够发现此文件种大部分字节要么是全1（白色像素点），要么是全0（黑色像素点）。
  2. 可以假设此加密文件的末尾仅包含 字节 0xFF（they form the bottom of the white frame that delimits the QR code）。给定文件的大小超过了 $410^6$，此文件的末尾的白色像素点的数量一定大于 $510^4$ 字节（最大的密钥长度）
  3. 我们可以假设异或运算是一组一组进行的，并且分组大小等于密钥长度。这就意味着加密文件的末尾存在一些分组，这些分组是由密钥和 0xFF异或而来。
  4. 因为最后一个分组可能不是完整的（如果密钥的长度不是文件长度的整数倍），所以我们要选择倒数第二个分组。我们发现最后一个分组的长度是文件的长度除以预期的密钥长度的余数。倒数第二个分组以一个字节开始，从文件末尾到它的距离等于密钥的预期长度加上最后一个块的长度。
  5. 为了了解我们是否在使用正确的密钥长度，我们可以先恢复密钥的前两个字节（通过将倒数第二个块的前两个字节和 两个 0xFF 异或）。如果我们将密钥的前两个字节和加密文件的前两个字节相异或，那么结果应该是 [0x42,0x4D]（bmp文件头）。
  6. 找到所有使得加密文件的前两个字节能够被正确解密的所有密钥长度。对于得到的值，通过将加密文件的倒数第二个块和 0xFF 异或 来重构出完整密钥。
  7. 使用找到的密钥解密文件。我们可以手动地查看所有恢复的文件并找到那个可以正常显示的 QR code图片。
  8. 扫描解密后的QR code 获取 flag
def xor(data: list, key: list) -> list:
    return [data[i] ^ key[i % len(key)] for i in range(len(data))]

with open('qr_xor_key.bmp', 'rb') as f:
    encrypt_data = list(f.read())

for key_len in range(2, 50001):
    last_part_len = (rem_len if (rem_len := len(encrypt_data) % key_len) else key_len)
    test_key_part = xor(encrypt_data[-last_part_len - key_len:-last_part_len - key_len + 2], [0xFF])
    header = xor(encrypt_data[:2], test_key_part)
    if header == [0x42, 0x4D]:
        find_key = xor(encrypt_data[-last_part_len - key_len:-last_part_len], [0xFF])
        with open('decrypt_qr_' + str(key_len) + '.bmp', 'wb') as f:
            f.write(bytearray(xor(encrypt_data, find_key)))

解题步骤

这其实是一个超级简单的问题，仅仅需要一点点线性代数的知识就可以做出来。设 三个非奇异矩阵 X，Y，Z，其中 Z的行列式值为1且 Y = X * Z 。根据相关知识可以知道 Z的行列式值等于 X的行列式值乘以 Y 的行列式值。想具体的了解可以看这篇文章：矩阵乘积的行列式等于行列式的乘积 |AB|=|A||B| - 知乎 (zhihu.com)

题目中恰好就生成了一个行列式值为 1 的 32 行 32 列的矩阵 Z，相关函数的定义在sage文档里写了：

https://doc.sagemath.org/html/en/reference/matrices/sage/matrix/special.html#sage.matrix.special.random_matrix

题目告诉我们如果 flag 的当前比特是 1，就让 X 乘以 Z 并赋值给 Y。根据上述知识可以知道当 X 和 Y 的行列式值相等的时候，我们就可以认定 X 乘以了 Z 并认为flag 当前比特是 1。那如果 X 和 Y的行列式值在不乘以Z的也相等呢？这种情况发生的概率很低，因为随机生成两个32行32列的矩阵，其行列式值一般都是不等的。

解题步骤

密钥是斐波那契数列的第N项。希望找到一个在具有p个元素的有限域上的多项式 $f_n$，使得 $f_n$ 和斐波那契数列的项具有某些关系。可以如下定义一个多项式 ：$f_n(x) = a_n + b_n * x$，也就是 $x^{n+1}$ 除以 $x^2-x-1$ 的余数，系数运算模p。我们发现：

$f_0(x) = 0 + 1 * x$

$f_1(x) = 1 + 1 * x$

$f_2(x) = 1 + 2 * x$

$f_3(x) = 2 + 3 * x$

可以发现当 $n>=1$，我们有如下公式：

$fn(x)=x*f{n-1}(x)\,mod\,(x^2-x-1)=b{n-1}+(a{n-1}+b_{n-1}\,mod\,p)*x$

这就表明 $an=F_n,b_n=F{n+1}$ ，$F_i$ 是斐波那契数列的第 i 项。

所以 求 $Fn$ 即是 求 $a_n$，也就是要求 $f{n}=x^{n+1}\,mod\,(x^2-x-1)$。

题目描述

from Crypto.Cipher import AES
from Crypto.Hash import SHA256
from secrets import token_hex


class CipherG(object):

    ROUNDS = 64
    S1 = [3, 5, 11, 12, 15, 7, 1, 13, 2, 0, 10, 9, 6, 14, 4, 8]
    S2 = [8, 12, 15, 13, 4, 5, 9, 1, 7, 3, 0, 2, 11, 10, 6, 14]

    def __init__(self, key: int):
        self.key = key
        self.key_schedule = [(self.key >> (24 - 8 * i)) & 0xFF for i in range(0, 4)]

    def _encrypt_round(self, x: int, k: int) -> int:
        u = x ^ k
        u1, u2 = (u >> 4) & 0x0F, u & 0x0F
        u1, u2 = self.S1[u1], self.S2[u2]
        u = ((u1 << 4) + u2) & 0xFF
        return ((u << 3) | (u >> 5)) & 0xFF

    def _encrypt_feistel(self, x: int) -> int:
        y1, y2 = (x >> 8) & 0xFF, x & 0xFF
        for i in range(0, self.ROUNDS):
            y1, y2 = y2, y1 ^ self._encrypt_round(y2, self.key_schedule[i % 4])
        return ((y2 << 8) + y1) & 0xFFFF

    def encrypt(self, x: bytearray) -> bytearray:
        x += bytearray([0x00] * (len(x) % 2))
        for i in range(0, len(x) // 2):
            current_block = ((x[i * 2] << 8) + x[i * 2 + 1]) & 0xFFFF
            current_block = self._encrypt_feistel(current_block)
            x[i * 2], x[i * 2 + 1] = (current_block >> 8) & 0xFF, current_block & 0xFF
        return x

解题步骤

很典型的中途相遇攻击的模型，之所以可以尝试这个操作是因为两个密钥都只有 28 比特（和 0xFFFFFFF相与了）

相关知识：https://en.wikipedia.org/wiki/Meet-in-the-middle_attack

预期在 $2^{29}$ 的时间内可以找到符合预期的密钥对。使用 C++ 花费了10分钟的时间..

#include <iostream>
#include <unordered_map>

using namespace std;

uint8_t ROUNDS = 64;
uint8_t S1[16] = {3, 5, 11, 12, 15, 7, 1, 13, 2, 0, 10, 9, 6, 14, 4, 8};
uint8_t S2[16] = {8, 12, 15, 13, 4, 5, 9, 1, 7, 3, 0, 2, 11, 10, 6, 14};

uint8_t encrypt_round(const uint8_t x, const uint8_t k)
{
    uint8_t u = x ^ k;
    uint8_t u1 = (u >> 4) & 0x0F, u2 = u & 0x0F;
    u1 = S1[u1];
    u2 = S2[u2];
    u = (u1 << 4) + u2;
    return ((u << 3) | (u >> 5));
}

uint16_t encrypt_feistel(const uint16_t x, const uint8_t k1, const uint8_t k2, const uint8_t k3, const uint8_t k4)
{
    uint8_t y1 = (x >> 8) & 0xFF, y2 = x & 0xFF;
    for (uint8_t i = 0; i < ROUNDS / 4; i += 1)
    {
        uint8_t tmp = y1;
        y1 = y2;
        y2 = tmp ^ encrypt_round(y2, k1);
        tmp = y1;
        y1 = y2;
        y2 = tmp ^ encrypt_round(y2, k2);
        tmp = y1;
        y1 = y2;
        y2 = tmp ^ encrypt_round(y2, k3);
        tmp = y1;
        y1 = y2;
        y2 = tmp ^ encrypt_round(y2, k4);
    }
    return (y2 << 8) + y1;
}

void encrypt_ecb(const uint32_t key, const uint16_t *in, const uint16_t in_len, uint16_t *out)
{
    uint8_t k1 = (key >> 24) & 0xFF, k2 = (key >> 16) & 0xFF, k3 = (key >> 8) & 0xFF, k4 = key & 0xFF;
    for (uint16_t i = 0; i < in_len; i += 1)
    {
        out[i] = encrypt_feistel(in[i], k1, k2, k3, k4);
    }
}

void decrypt_ecb(const uint32_t key, const uint16_t *in, const uint16_t in_len, uint16_t *out)
{
    uint8_t k1 = (key >> 24) & 0xFF, k2 = (key >> 16) & 0xFF, k3 = (key >> 8) & 0xFF, k4 = key & 0xFF;
    for (uint16_t i = 0; i < in_len; i += 1)
    {
        out[i] = encrypt_feistel(in[i], k4, k3, k2, k1);
    }
}

int main ()
{
    const uint16_t pt[4] = {0x65fe, 0x13fe, 0xd92a, 0xdbc9}; // pt_str from task.txt
    const uint16_t ct[4] = {0x89b9, 0xf2c7, 0xe6dd, 0x32f8}; // ct_str from task.txt
    
    uint16_t tmp[4] = {0x0000, 0x0000, 0x0000, 0x0000};
    unordered_map<uint64_t, uint32_t> key_map;

    cout << "Part 1 (step 4) start!" << endl;
    for (uint32_t key1 = 0;  key1 < 0x0FFFFFFF; key1 += 1)
    {
        encrypt_ecb(key1, pt, sizeof(pt) / sizeof (uint16_t), tmp);
        uint64_t res_encr = ((uint64_t)(tmp[0]) << 48) + ((uint64_t )(tmp[1]) << 32) + ((uint64_t)(tmp[2]) << 16) + tmp[3];
        key_map[res_encr] = key1;
    }

    cout << "Part 2 (step 5) start!" << endl;
    for (uint32_t key2 = 0;  key2 < 0x0FFFFFFF; key2 += 1)
    {
        decrypt_ecb(key2, ct, sizeof(pt) / sizeof (uint16_t), tmp);
        uint64_t res_decr = ((uint64_t)(tmp[0]) << 48) + ((uint64_t )(tmp[1]) << 32) + ((uint64_t)(tmp[2]) << 16) + tmp[3];
        if (auto iter = key_map.find(res_decr); iter != key_map.end())
        {
            cout << "Find possible key pair: [" << iter->second << ", " << key2 << "]" << endl;
        }
    }

    cout << "Done!" << endl;
    return 0;
}
