from Crypto.Util.number import getPrime, long_to_bytes, bytes_to_long

with open("flag.txt", "rb") as f:
    FLAG = f.read()

n = bytes_to_long(FLAG)

#make sure i have a big modulus
while n.bit_length() < 2048:
    n *= n

def encrypt(m1, m2):
    e = getPrime(256)
    assert m1.bit_length() >= 1600 and long_to_bytes(m1).startswith(b"SEE{"), 'first message must be at least 1600 bits and begin with "SEE{"'
    assert 500 <= m2.bit_length() <= 600, 'second message must be within 500 to 600 bits'

    return pow(m1, e, n), pow(m2, e, n)


def main():
    try:
        m1 = int(input("Message 1 (as integer) : ").strip())
        m2 = int(input("Message 2 (as integer) : ").strip())
        c1, c2 = encrypt(m1, m2)
        print(f"\nCiphers: \n{[c1,c2]}")
    except Exception as e:
        print(e)

if __name__ == '__main__':
    main()

RSA 选择明密文攻击 - CTF Wiki

解题思路是构造具有特定关系的多组明文对m1 m2 然后求出多组密文对的最大公因数，在本题中公因数大概就是n的幂，随后枚举幂数开根，能够开根出来的就是n了

$$c_1 \equiv m_1^e \mod n\c_2 \equiv m_2^e \mod n\观察这题的m_1 \space m_1的大小，m_2的平方是小于m_1的，所以构造m_2^3 = m1比较合适\所以我们有c_1^3 \equiv (m_1^e)^3 \mod n \space 即 c_2^3 \equiv m_1^e \mod n \space 所以c_1 - c_2^3 \equiv 0 \mod n\即 c_2^3 - c_1 = kn(等式1)\我们可以构造多个具有上述关系的m_1 \space m_2 然后得到多个等式1，最后求这些等式的最大\公因数即可拿到n的幂(因为题目说当n的比特位数小于2048就要乘上自己)$$

其实构造具有上述关系的m1 m2 也是需要一点小技巧的，特别是还要求m1以特定字符串开头

from pwn import *
from Crypto.Util.number import *
import os
import gmpy2
import ast


def gen():
    #下面三句是构造方法
    m1 = bytes_to_long(b"SEE{" + os.urandom(1700 // 8 - 4))
    m2 = int(gmpy2.iroot(m1, 3)[0])#m1不一定是立方数，所以m2应是m1的立方根向下取整
    m1 = m2**3#巧妙地是m2的三次方的高位4字节依旧是SEE{

    io = remote("fun.chall.seetf.sg", 30004)
    io.sendlineafter(b"Message 1 (as integer) : ", str(m1).encode())
    io.sendlineafter(b"Message 2 (as integer) : ", str(m2).encode())
    io.recvuntil(b"Ciphers: \n")
    c1, c2 = ast.literal_eval(io.recvlineS().strip())
    return c2**3 - c1


x = gcd(gcd(gen(), gen()), gen())
for i in range(2, 10):
    r, exact = gmpy2.iroot(x, i)
    if exact:
        print(i, long_to_bytes(r))

import math, json
from secrets import token_urlsafe
from Crypto.Util.number import isPrime, bytes_to_long, long_to_bytes

def main():
    try:
        # create token
        token = token_urlsafe(8)
        js = json.dumps({'token': token})
        
        # select primes
        print(f'Welcome to the RSA testing service. Your token is "{token}".')
        print('Please give me 128-bit primes p and q:')
        p = int(input('p='))
        assert isPrime(p) and p.bit_length() == 128, 'p must be a 128-bit prime'
        assert str(float(math.pi)) in str(float(p)), 'p does not look like a certain universal constant'
        q = int(input('q='))
        assert isPrime(q) and q.bit_length() == 128, 'q must be a 128-bit prime'
        assert str(float(math.e)) in str(float(q)), 'q does not look like a certain universal constant'

        # encode
        print('We will use e=65537 because it is good practice.')
        e = 65537
        m = bytes_to_long(js.encode())
        c = pow(m, e, p * q)

        # decode
        print('Now what should the corresponding value of d be?')
        d = int(input('d='))
        m = pow(c, d, p * q)
        
        # interpret as json
        js = json.loads(long_to_bytes(m).decode())
        assert js['token'] == token, 'Invalid token!'
        print('RSA test passed!')

        if 'flag' in js:
            from secret import flag
            print(flag)
    
    except Exception as e:
        print(e)

if __name__ == '__main__':
    main()

阅读题目源代码就可以发现我们输入的p跟q转换成float后必须包含pi跟e，这一部分直接暴力求解

第二部分是要找出使得$$m’ \equiv (m^e)^d \mod p*q \space的\space d (等式一)$$,其中m是原来的token编码后的值，而m’是修改后的值，为了拿到flag，m’应当是这样的形式:

js2 = json.dumps({"token": token, "flag": 1})

我们的目标是求d，显然地这是一个离散对数问题DLP。尝试使用Pohlig-Hellman算法，此算法是一种求解光滑阶循环群上的离散对数的方法。而等式一等价于如下等式：

$$m’ \equiv (m^e)^d \mod p \ m’ \equiv (m^e)^d \mod q $$

因为p q都是128bit的素数，所以上述两个式子可以看成是在有限域p和有限域q上的运算；如果有限域p和有限域q具有光滑阶数(有限域已经是循环群了，所以只需要满足光滑阶条件)，那么我们就可以运用Pohlig-Hellman算法求出两个等式的d，然后运用中国剩余定理求出在模p*q下的d，然后就可以得到flag。

所以现在需要构造两个128比特的包含pi和e的而且具有光滑阶数的素数p和q

p和q具有光滑阶数即p-1 和 q-1(模素数p的循环群阶数是p-1) 是光滑数，所以p-1和q-1可以因数分解为小素数的乘积，所以我们需要编写代码寻找满足这样条件的p q，找到了之后在运用Pohlig-Hellman算法求解DLP问题即可。不过我们也不需要手写Pohlig-Hellman算法，Sage自带了；具体实现链接可以看这个：离散对数

from pwn import *
from Crypto.Util.number import *
from sage.all import GF, crt, factor
import json

context.log_level = "debug"


def nextPrime(x):
    x += 1
    while not isPrime(x):#当x不是素数
        x += 1
    return x


def smoothEnough(x):
    fs = factor(x) #分解得到x的质因数
    #factor(10) = 2 * 5
    # all函数 检查x的所有质因数p是否都小于2的40次方,_是*号,p才是质因数
    return all([p < 2**40 for p, _ in fs])


def findSmooth(x):
    p = nextPrime(x)
    while not smoothEnough(p - 1): #当p-1存在大于2的40次方的质因数
        p = nextPrime(p)#求p的下一个素数并赋值给p
    return p


io = remote("fun.chall.seetf.sg", 30002)
io.recvuntil(b'token is "')#接收信息到 token is "为止
token = io.recvuntilS(b'"')[:-1]#接收信息到 " 为止
#上面两句话就是获取 token is "xxxxx" 这个信息的双引号之间的值，也就是token值
print(token)
js = json.dumps({"token": token})
js2 = json.dumps({"token": token, "flag": 1})

p = findSmooth(314159265358979300000000000000000000000)
q = findSmooth(271828182845904500000000000000000000000)

while True:
    print(float(p), float(q))

    g = pow(bytes_to_long(js.encode()), 65537, p * q) #c
    #n最多32字节，我们手动添加了一个key为flag，为了encode之后不超过32字节，所以要删去多余空格
    y = bytes_to_long(js2.replace(" ", "").encode()) % (p * q)#m'
    try:
        dp = GF(p)(y).log(g) # 计算 g^dp = y mod p 的dp
        dq = GF(q)(y).log(g) # 计算 g^dq = y mod q 的dq
        #运用中国剩余定理求 g^d = y mod p*q 的d
        d = int(crt([int(dp), int(dq)], [p - 1, q - 1]))
    except Exception as ex: #如果当前选择的p q 求不出来，选择别的继续求解
        print(ex)
        p = findSmooth(p) #寻找光滑数
        q = findSmooth(q)
        continue
    break
#循环结束，找到合适的p q d，发送到服务器获得flag
print(p, q, d)
io.sendlineafter(b"p=", str(p).encode())
io.sendlineafter(b"q=", str(q).encode())
io.sendlineafter(b"d=", str(d).encode())
io.interactive()

The True ECC

# python ecc.py > out.py
#ecc.py
from random import randint
from os import urandom
from Crypto.Cipher import AES
from Crypto.Util.Padding import pad, unpad
from hashlib import sha1

from typing import Tuple


class Ellipse:

    """Represents the curve x^2 + ay^2 = 1 mod p"""

    def __init__(self, a: int, p: int):

        self.a = a
        self.p = p

    def __repr__(self) -> str:
        return f"x^2 + {self.a}y^2 = 1 mod {self.p}"

    def __eq__(self, other: 'Ellipse') -> bool:
        return self.a == other.a and self.p == other.p

    def is_on_curve(self, pt: 'Point') -> bool:

        x, y = pt.x, pt.y
        a, p = self.a, self.p
        return (x*x + a * y*y) % p == 1


class Point:

    """Represents a point on curve"""

    def __init__(self, curve: Ellipse, x: int, y: int):

        self.x = x
        self.y = y
        self.curve = curve
        assert self.curve.is_on_curve(self)

    def __repr__(self) -> str:
        return f"({self.x}, {self.y})"

    def __add__(self, other: 'Point') -> 'Point':

        x, y = self.x, self.y
        w, z = other.x, other.y
        a, p = self.curve.a, self.curve.p

        nx = (x*w - a*y*z) % p
        ny = (x*z + y*w) % p
        return Point(self.curve, nx, ny)

    def __mul__(self, n: int) -> 'Point':

        assert n > 0

        Q = Point(self.curve, 1, 0)
        while n > 0:
            if n & 1 == 1:
                Q += self
            self += self
            n = n//2
        return Q

    def __eq__(self, other: 'Point') -> bool:
        return self.x == other.x and self.y == other.y


def gen_secret(G: Point) -> Tuple[Point, int]:

    priv = randint(1, p)
    pub = G*priv
    return pub, priv


def encrypt(shared: Point, pt: bytes) -> bytes:

    key = sha1(str(shared).encode()).digest()[:16]
    iv = urandom(16)
    cipher = AES.new(key, AES.MODE_CBC, iv)
    ct = cipher.encrypt(pad(pt, 16))
    return iv + ct


def decrypt(shared: Point, ct: bytes) -> bytes:

    iv, ct = ct[:16], ct[16:]
    key = sha1(str(shared).encode()).digest()[:16]
    cipher = AES.new(key, AES.MODE_CBC, iv)
    pt = cipher.decrypt(ct)
    return unpad(pt, 16)
