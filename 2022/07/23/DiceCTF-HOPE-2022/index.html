#!/usr/local/bin/python -u

from Crypto.Cipher import AES
from Crypto.Random import get_random_bytes
from Crypto.Util.Padding import pad
from Crypto.Util.strxor import strxor
from more_itertools import ichunked

BLOCK = AES.block_size
FLAG = open('flag.txt', 'rb').read().strip()

def encrypt_block(k, pt):
  cipher = AES.new(k, AES.MODE_ECB)
  return cipher.encrypt(pt)

def encrypt(k, pt):
  assert len(k) == BLOCK
  pt = pad(pt, BLOCK)
  ct = b''
  for bk in ichunked(pt, BLOCK):
    ct += strxor(encrypt_block(k, k), bytes(bk))
  return ct

def main():
  k = get_random_bytes(BLOCK)
  enc = encrypt(k, FLAG)
  print(f'> {enc.hex()}')

  pt = bytes.fromhex(input('< '))[:BLOCK]
  enc = encrypt(k, pt)
  print(f'> {enc.hex()}')

if __name__ == '__main__':
  main()

	print("Clearly, you must already know the flag!")

	with open('flag.txt','rb') as f:

		print(flag.decode())

else:
	print("hack harder")

这是在问这么一个问题：任意选择n、e、p、q、d，求对未知的密文c，其解密出的m的格式是hope{xxxx}

用数学表达式表达为：$m’\equiv c^d \mod n $，其中$m’$ 要符合题目给的regex。我们可以随便取个符合regex的$m’$，比如hope{a}，注意字节顺序。

然后这个问题就变成了离散对数问题，所以我们只要让p-1,q-1 都够smooth，然后分别解dlog之后crt就能求得需要得e了。

replacement

题如其名，高级的替代密码，将打乱的明文告诉了我们。关键就在于找出密文跟明文的对应关系。而且本题还加大了难度，映射表有41个字符…..

import random

with open('text.txt') as f:
    plaintext = f.read()

with open('flag.txt') as f:
    plaintext += '\n' + f.read()

characters = set(plaintext) - {'\n'}

shuffled = list(characters)
random.shuffle(shuffled)

replacement = dict(zip(characters, shuffled))

ciphertext = ''.join(replacement.get(c, c) for c in plaintext)

with open('output.txt', 'w') as f:
    f.write(ciphertext)

small-fortune

from Crypto.Util.number import *
from gmpy2 import legendre

flag = bytes_to_long(open("flag.txt", "rb").read())

p, q = getPrime(256), getPrime(256)
n = p * q

x = getRandomRange(0, n)
while legendre(x, p) != -1 or legendre(x, q) != -1:
    x = getRandomRange(0, n)

def gm_encrypt(msg, n, x):
    y = getRandomRange(0, n)
    enc = []
    while msg:
        bit = msg & 1
        msg >>= 1
        enc.append((pow(y, 2) * pow(x, bit)) % n)
        y += getRandomRange(1, 2**48)
    return enc

print("n =", n)
print("x =", x)
print("enc =", gm_encrypt(flag, n, x))

首先有256bits的$n=pq$ 和一个随机数$x$（二次剩余的部分好像用不到）当作公钥，然后加密流程是先生成一个未知的随机数$y$，之后bit bit bit 加密：

$c_i \equiv y_i^2x^{b_i}(\mod n)$

其中的$y0=y,y_i=y{i-1}+d_i$ ，且 $d_i$是$2^{48}$以内的随机数。

首先是假定flag.txt里面的flag格式是hope{....}且在最后没有换行，那么我们知道$b_0=1,b_1=0$。从这边可以列出两个等式：

$c_0\equiv y^2x^{b_0}\equiv y^2x \space(\mod n)$

$c1\equiv (y+d1)^2x^{b_0}\equiv(y+d_1)^2 (\mod n)$

分别移项可以得到两个多项式$f(y,d_1),g(y,d_1)$，用resultant可得$h=Res_y(f,g)$是个$d_1$的四次多项式。 0
