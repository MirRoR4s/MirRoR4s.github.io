import os

flag = "XXXX"
some_simple_text = "YYYY"
key = os.urandom(43)

def encrypt(plain_text):
    assert len(plain_text) == len(key)
    ciphertext = bytearray([a ^ b for a, b in zip(plain_text, key)])
    return ciphertext.hex()

print(encrypt(flag.encode("utf-8")))
print(encrypt(some_simple_text.encode("utf-8")))

先尝试用已知明文accessdenied{恢复出一部分密钥来。

之后写脚本，一个字母一个字母地猜就可以

from functools import reduce


def xor(*args):
    if len(args) == 2:
        x, y = args
        return bytes([a ^ b for a, b in zip(x, y)])
    return reduce(xor, args)


ct = [
    bytes.fromhex('65f32f851cdb20eee875eea5a9a30f826cfd247eb550dcc89d1d4cdf952f5c28ca5f162355567fd262bb96'),
    bytes.fromhex('70f8259330c137d4e873ff9ea6a559ab2dea1a60d943859aa545578395301d28a0741d1e065a24d45cb19f')

]
key = xor(b"accessdenied{", ct[0])



while True:
    pts = [xor(key, c) for c in ct]
    for i, pt in enumerate(pts):
        print(i, pt)
    idx = int(input("Enter index: "))
    if idx == -1:
        break
    c = input("Enter next char: ")[0]
    key += bytes([ord(c) ^ ct[idx][len(key)]])
    print("Current key:", key.hex())
print([xor(key, c) for c in ct])
