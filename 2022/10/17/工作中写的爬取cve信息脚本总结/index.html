class Spider:
    def __init__(self,url=""):
        """

        :param url: url是一个汇总网址，里面记录以CVEid的形式记录了很多CVE的信息
        """
        self.url = url

    def get_info(self):
        # 获取到所有CVE的链接并写入文件中
        file = open('./CVE_changes.txt', 'w', encoding='utf-8')
        url = base_url  # 这是一个汇总了每天更新的cve-id的网站
        burp0_headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/104.0.5112.81 Safari/537.36",
            "Accept": "text/html,application/xhtml+xml,application/xml;q=0.9,image/avif,image/webp,image/apng,*/*;q=0.8,application/signed-exchange;v=b3;q=0.9",
            "Accept-Language": "zh-CN,zh;q=0.9",
            "Accept-Encoding": "gzip, deflate", "Connection": "close",
        }

        r = requests.get(url=url, headers=burp0_headers)
        soup = BeautifulSoup(r.text, 'html.parser')

        for link in soup.find_all('a'):
            tmp = link.get('href')
            if 'CVE' in tmp:
                file.write(tmp)
                file.write('\n')

        # 以cve页面为key，日期为value建立字典
        pattern = re.compile(r'((CVE|today)(\.\d{4})*(\.\d{2})*(\.\d{2})*\.(html|tar))|(\d{4}-\d{2}-\d{2})')
        database = []
        for td in soup.find_all('td'):
            data = (pattern.match(td.text))
            if data is not None:
                # print(data.group())
                database.append(data.group())

        for i in range(0, len(database) - 1, 2):
            dict[database[i]] = database[i + 1]  # 以cve页面为key，日期为value建立字典

        file.close()

    def get_tar_info(self):
        # 从CVE-id.txt读取
        file = open('./CVE-id.txt', 'w', encoding='utf-8')
        r = requests.get(url=self.url)
        soup = BeautifulSoup(r.text, 'html.parser')

        for link in soup.find_all('a'):
            # print(link.get('href'))
            file.write(link.get('href'))
            file.write('\n')

        file.close()

    def run(self,target):
        workbook = xlsxwriter.Workbook(filename='./{}.xlsx'.format(target),
                                       options={  # 全局设置
                                           'strings_to_numbers': True,  # str 类型数字转换为 int 数字

                                           'constant_memory': True,  # 连续内存模式 (True 适用于大数据量输出)
                                           'default_format_properties': {
                                               'font_name': '微软雅黑',  # 字体. 默认值 "Arial"
                                               'font_size': 10,  # 字号. 默认值 11
                                               # 'bold': False,  # 字体加粗
                                               # 'border': 1,  # 单元格边框宽度. 默认值 0
                                               'align': 'left',  # 对齐方式
                                               'valign': 'vcenter',  # 垂直对齐方式
                                               'text_wrap': True,  # 单元格内是否自动换行
                                               # ...
                                           },
                                       })  # 创建一个Excel文件 worksheet

        worksheet = workbook.add_worksheet()  # Sheet1
        fmt = workbook.add_format({'align': 'center', 'valign': 'vcenter', 'fg_color': 'green', 'font_size': 10})

        # sheet.merge_range(x1, y1, x2, y2, value, cell_format=None)

        bold = workbook.add_format({'bold': True, 'fg_color': 'orange', 'valign': 'vcenter', 'align': 'center'})
        worksheet.write('A2', 'CVE-ID', bold)  # 在A1单元格写入'Hello'字符串

        worksheet.write('B2', 'Description', bold)  # 在B1单元格写入'World'字符串
        # worksheet.write('C2', 'State', bold)
        # worksheet.write('D2', 'Problem Types', bold)
        worksheet.write('C2', 'Vendors', bold)
        worksheet.write('D2', 'Products', bold)
        worksheet.write('E2', 'Versions', bold)
        # worksheet.write('H2', 'References', bold)
        # worksheet.write('G2', 'Date', bold)

        worksheet.set_column('B:B', 70)  # 设置C到D列单元格宽度为20像素
        worksheet.set_column('C:H', 20)

        f = open('./CVE-id.txt', 'r').readlines()
        row = 3
        for i in tqdm(f):
            # print('url=', str(i))

"""读取公司组件的xlsx格式文件"""
def read_excel(name):
    wb = openpyxl.load_workbook(name)
    # 获取行数和列数
    sheet = wb['Sheet1']
    max_row = sheet.max_row
    dic = {}
    # 遍历excel数据
    for i in range(3, max_row + 1):
        key = (sheet.cell(i, 3).value).lower()
        tmp = []
        for j in range(4, 5):
            value = str(sheet.cell(i, j).value)
            if value:
                tmp += re.split('、', value)
        dic[key] = tmp

    return dic

"""该类负责对10月份的cve信息进行解析，传入的参数是爬取好的json格式数据"""
class  parse:

    def __init__(self,req_data):
        """

        :param req_data: req_data是Req类的实例，
        """
        self.data = req_data.send_req()

    def get_cve_id(self):  # 获取CVE_ID
        try:
            return self.data['cveMetadata']['cveId']
        except Exception as error:
            print('get_cve_id发生错误：',error)
            return "N/A"

    def get_cve_desc(self):  # 获取CVE的描述信息
        try:
            return self.data['containers']['cna']['descriptions'][0]['value']
        except Exception as error:
            print(error)
            return "N/A"

    def get_cve_state(self):  # 获取CVE的state
        try:
            return self.data['cveMetadata']['state']
        except Exception as error:
            print(error)
            return "N/A"

    def get_cve_prob_type(self):  # 获取CVE的问题类型信息
        try:
             return self.data['containers']['cna']['problemTypes'][0]['descriptions'][0]['description']

        except Exception as error:
            print(error)
            return "N/A"

    def get_cve_ventor(self):
        try:
            return  self.data['containers']['cna']['affected'][0]['vendor']
        except Exception as error:
            print(error)
            return "N/A"

    def get_cve_version(self):
        try:
            return (self.data['containers']['cna']['affected'][0]['versions'][0]['version']) + ','+ ' ' + self.data['containers']['cna']['affected'][0]['versions'][0]['status']
        except:
            #print('get_cve_version未有此项')
            try:
                return self.data['containers']['cna']['affected'][0]['versions'][0]['version']
            except:
                pass

            return "N/A"


    def get_cve_product(self):
        try:
           return self.data['containers']['cna']['affected'][0]['product']
        except:
            return "N/A"

    def get_cve_references(self):
        info = """"""
        #print(data)
        try:
            array = self.data['containers']['cna']['references']
            print(array)
            # print(array)
            for i in array:
              info += 'URL'+ ":" + i['url'] + '\n\n'

            info += "View additional information about https://nvd.nist.gov/view/vuln/detail?vulnId={} on NVD.(Note: The NVD is not operated by the CVE Program)\n".format(
                self.get_cve_id())
        except Exception as error:
            print('发生异常',error)
            return info
        return info



"""该类负责获取输出并控制输出的格式，目前只有匹配公司组件的内容才输出"""
class ControlOutputFormat:

    """"""
    def __init__(self,cve10,row,worksheet,name='test.xlsx'):
        """

        :param cve10: 一个parse类的实例
        :param row:  表示写入xlsx表格的第几行
        :param worksheet: 一个xlsx表格
        :param name: 要打开的含有公司组件的xlsx表格的文件名
        """
        self.flag = False #flag是标志位，表示当前的内容是否含有公司的组件
        self.row = row
        self.id = cve10.get_cve_id()
        self.desc = cve10.get_cve_desc()
        self.state = cve10.get_cve_state()
        self.pb = cve10.get_cve_prob_type()
        self.ven = cve10.get_cve_ventor()
        self.pro = (cve10.get_cve_product()).lower()
        self.ver = cve10.get_cve_version()
        self.read_excel = read_excel(name)
        self.worksheet = worksheet
        self.dic =  {0:'A',1:'B',2:'C',3:'D',4:'E',5:'F',6:'G',7:'H'}



    """检测内容中是否有和公司匹配的组件"""
    def sol_con(self):

        if self.read_excel.get(self.pro) != None:
            print('发现与公司组件配对漏洞信息')
            self.flag = True

    def output(self):
        self.row = str(self.row)
        ans =  [self.id, self.desc,  self.ven, self.pro, self.ver]
        self.sol_con()


        if self.flag:
            for i in range(len(ans)):
                # print(i)
                self.worksheet.write(self.dic[i] + self.row, str(html.unescape(ans[i])))
            return self.flag

该脚本由Req、parse、ControlOutputFormat类组成。Req负责发起爬虫请求。parse负责对请求之后的数据进行解析，最后由ControlOutputFormat对解析之后的数据进行输出的判断（匹配公司组件才输出）

import smtplib
from email.mime.text import MIMEText
from email.header import Header
from email.mime.application import MIMEApplication
from email.mime.multipart import MIMEMultipart

class SendEmail:
    def __init__(self,smtpserver,user,password,sender,receiver,subject,content):
        """

        :param smtpserver: smtp服务器的ip地址，可以是域名
        :param user: 登录邮箱的用户名
        :param password: 登录邮箱的密码
        :param sender: 发送邮件时显示的发送方
        :param receiver: 发送邮件时显示的接收方
        :param subject:  发送邮件的主题
        :param content:  发送邮件的内容
        """
        self.smtpserver = smtpserver
        self.user = user
        self.password = password
        self.sender = sender
        self.receiver = receiver
        self.subject = subject
        self.content = content

    def send(self,accessory,content_type,content_disposition):
        """
        发送邮件,支持带附件
        accessory是一个所打开附件的文件流
        content_type指明发送附件的内容类型
        content_dispistion 指明发送附件的文件名等
        :return: 无
        """
        att = MIMEText(accessory, "base64", 'utf-8')
        #att['Content-Type'] = 'application/zip'
        #att['Content-Disposition'] = 'attachment;filename="1.xlsx"'
        att['Content-Type'] = content_type
        att['Content-Disposition'] = content_disposition

        msgRoot = MIMEMultipart()
        msgRoot.attach(MIMEText(content, 'html', 'utf-8'))
        msgRoot['Subject'] = self.subject
        msgRoot['From'] = self.sender
        msgRoot['To'] = self.receiver
        msgRoot.attach(att)

        '''msg['From'] = "huangjiantao@damddos.com"
        msg['To'] = "790356373@qq.com"
        msg['Subject'] = Header(subject,'utf-8')'''

        # SSL协议端口号要使用465
        smtp = smtplib.SMTP_SSL(smtpserver, 465)

        # HELO向服务器标志用户身份
        smtp.helo(smtpserver)

        # 服务器返回结果确认
        smtp.ehlo(smtpserver)

        # 登录邮箱服务器用户名密码
        smtp.login(user, password)

        print("Send email start...")
        smtp.sendmail(sender, receive, msgRoot.as_string())
        smtp.quit()
        print("email send end!")
