数据结构C语言版_线性表的顺序存储结构相关代码实现

C/C++语言实现线性表的顺序存储结构，附带 增 删 改 查。不要问为什么c跟c++混一起，书上的引用也是c++的，所以说不要拘泥于语言啦

0x01 线性表的定义#

为了尽量跟书上一致，就采用书上的写法。

typedef struct
{
    ElemType *elem;
    int length;
    int listsize;
} SqList;

0x02 线性表的初始化#

bool InitList_Sq(SqList &L)
{
    L.elem = (ElemType*)malloc(LIST_INIT_SIZE*sizeof(ElemType));
    if(!L.elem)
    {
        return false;
    }
    L.length = 0;
    L.listsize = LIST_INIT_SIZE;
    return true;
}

0x03 线性表的元素插入#

插入的位置是说第几个位置插入，比如第一个位置，第二个位置。返回值为true表示插入成功

bool ListInsert_Sq(SqList &L,int i,ElemType e)
{

    if(i<1 || i>L.length+1)//插入位置不合法
    {
        return false;
    }
    if(L.length >= L.listsize)//判断当前线性表的存储空间是否足够
    {
        ElemType *newbase = (ElemType*)realloc(L.elem,(L.listsize+LISTINCREMENT)*sizeof(ElemType));
        if(!newbase)
        {
            return false;//分配失败，退出
        }
        L.elem = newbase;
        L.listsize+= LISTINCREMENT;
    }
    int *q = &(L.elem[i-1]);//第i个位置的元素下标是i-1，获取此元素在线性表中的地址
    for(int* p=&(L.elem[L.length-2]); p>=q; p--) //最后一个元素的下标是L.length-1
    {
        *(p+1) = *p;
    }
    *q = e;
    ++L.length;
    return true;
}

0x04 线性表的元素删除#

返回值是删除的元素的值

int ListDelete_Sq(SqList &L,int i,ElemType e)
{

    if(i<1||i>L.length)
    {
        throw invalid_argument("删除元素的位置不合法");
    }
    int *p = &(L.elem[i-1]); //p指向要被删除元素的位置
    int *q = L.elem+L.length-1;//elem是基地址，q的结果是末尾指针的位置。
    for(++p; p<=q; p++)
    {
        *p = *(p+1);
    }

    --L.length;
    e = *q;
    return e;
}

0x05 线性表的查找#

未找到返回-1

int LocateElem(SqList &L,ElemType e)//查找线性表中第一个值与e相同的元素，并返回其位序，注意不是下标
{
    int *p = L.elem;
    int *q = L.elem + L.length-1;
    int cnt = 1;//记录位序
    while(p<=q)
    {
        if(*p==e)
        {
            return cnt;
        }
        ++cnt;
        ++p;
    }
    return -1;
}

0x06 打印线性表中元素#

void Print(SqList &L)
{
    int *p = L.elem;
    int *q = L.elem+L.length-1;
    while(p<=q)
    {
        cout<<*p<<" ";
        p++;
    }
    cout<<endl;
}

0x07 总体代码+测试#

#include<bits/stdc++.h>
#define LIST_INIT_SIZE 100 //线性表存储空间的初始分配量
#define LISTINCREMENT 10 //增量
#define ElemType int //int的别名为ElemType
using namespace std;


typedef struct
{
    ElemType *elem;
    int length;
    int listsize;
} SqList;
bool InitList_Sq(SqList &L)
{
    L.elem = (ElemType*)malloc(LIST_INIT_SIZE*sizeof(ElemType));
    if(!L.elem)
    {
        return false;
    }
    L.length = 0;
    L.listsize = LIST_INIT_SIZE;
    return true;
}
bool ListInsert_Sq(SqList &L,int i,ElemType e)
{

    if(i<1 || i>L.length+1)//插入位置不合法
    {
        return false;
    }
    if(L.length >= L.listsize)//判断当前线性表的存储空间是否足够
    {
        ElemType *newbase = (ElemType*)realloc(L.elem,(L.listsize+LISTINCREMENT)*sizeof(ElemType));
        if(!newbase)
        {
            return false;//分配失败，退出
        }
        L.elem = newbase;
        L.listsize+= LISTINCREMENT;
    }
    int *q = &(L.elem[i-1]);//第i个位置的元素下标是i-1，获取此元素在线性表中的地址
    //感觉书上好奇怪啊一会用括号+下标，下面循环又用指针，虽然本质一样但是形式不统一- -
    for(int* p=&(L.elem[L.length-2]); p>=q; p--) //最后一个元素的下标是L.length-1
    {
        *(p+1) = *p;
    }
    *q = e;
    ++L.length;
    return true;
}
int ListDelete_Sq(SqList &L,int i,ElemType e)
{

    if(i<1||i>L.length)
    {
        throw invalid_argument("删除元素的位置不合法");
    }
    int *p = &(L.elem[i-1]); //p指向要被删除元素的位置
    int *q = L.elem+L.length-1;//elem是基地址，q的结果是末尾指针的位置。
    for(++p; p<=q; p++)
    {
        *p = *(p+1);
    }

    --L.length;
    e = *q;
    return e;
}
int LocateElem(SqList &L,ElemType e)//查找线性表中第一个值与e相同的元素，并返回其位序，注意不是下标
{
    int *p = L.elem;
    int *q = L.elem + L.length-1;
    int cnt = 1;//记录位序
    while(p<=q)
    {
        if(*p==e)
        {
            return cnt;
        }
        ++cnt;
        ++p;
    }
    return -1;
}
SqList MergeList(SqList La,SqList Lb,SqList &c);//合并两个有序的线性表，注意是有序
void Print(SqList &L)
{
    int *p = L.elem;
    int *q = L.elem+L.length-1;
    while(p<=q)
    {
        cout<<*p<<" ";
        p++;
    }
    cout<<endl;
}
int main()
{
    SqList a;//定义线性表
    bool flag = InitList_Sq(a);//初始化刚刚定义的线性表
    if(!flag)cout<<"初始化失败\n";
    cout<<"线性表当前长度为:"<<a.length<<endl;
    cout<<"线性表的容量为:"<<a.listsize<<endl;
    for(int i=0;i<10;i++)ListInsert_Sq(a,i+1,i);//向线性表中插入10个元素
    Print(a);//打印线性表中的元素
    int e;
    e = ListDelete_Sq(a,1,e);//删除线性表中第一个元素
    Print(a);
    e = ListDelete_Sq(a,a.length,e);//删除线性表中最后一个元素
    Print(a);
    int loc = LocateElem(a,3);//查找线性表中值为3的元素的位序
    if(!loc)cout<<"查找的元素不在表中"<<endl;
    else cout<<loc<<endl;
}
